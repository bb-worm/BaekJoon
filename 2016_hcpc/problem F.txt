#include <stdio.h>

int main() {


	int i, j;
	int TC;
	
	int n;
	char *report;
	int *askii;

	scanf("%d", &TC);
	
	for (i = 0; i < TC; i++) {

		scanf("%d", &n);
		report = (char*)malloc(sizeof(char)*n);
		askii = (int*)malloc(sizeof(int) * 128);
		for (j = 0; j < 128; j++)
			askii[j] = 0;

		scanf("%s", report);

		for (j = 0; j < n; j++) {
			askii[report[j]]++;
		}

		for (j = 0; j < 128; j++) {
			if (askii[j]) {
				printf("%c %d\n", j, askii[j]);
			}
		}
		//free(report);
		//free(askii);
	}
	
	return 0;
	
}
