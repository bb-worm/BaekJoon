#include <stdio.h>

int check_str(char test[]);
char str[7];

int main() {

	char test[14];
	int N;
	int i;

	scanf("%s", str);
	scanf("%d", &N);

	for (i = 1; i <= N; i++) {
		scanf("%s", test);
		if (check_str(test))
			printf("0\n");
		else
			printf("%d\n", i);
	}
}

int check_str(char test[]) {
	int i, j;
	int H_Point;

	for (i = 0; i < 7; i++) {
		if (test[i] == 'H') {
			H_Point = i;
			break;
		}
		else if (test[i] == 'h')
			return 0;
	}
	i = 7;
	for (j = 0; j < H_Point; j++) {
		test[i++] = test[j];
	}
	
	for (j = 0; j < 7; j++) {
		if (str[j] != test[H_Point++])
			return 0;
	}
	return 1;

}
